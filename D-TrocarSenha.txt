#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/wait.h>

int main(int argc, char *argv[]){
pid_t son;
int status;

son = fork();

if(son == 0){
	char user[50];
	printf("Usuário: ");
	scanf("%s", user);
	execlp("sudo", "sudo", "passwd", user, NULL);
	perror("execlp");
	exit(EXIT_FAILURE);
}
else if(son < 0){
	perror("execlp");
	exit(EXIT_FAILURE);
}
else{
	wait(&status);
	if(WIFEXITED(status))
		fprintf(stdout, "\nComando executado com sucesso.\n\n");
	else
		fprintf(stdout, "\nComando nao executado.\n\n");
}
return 0;
}
